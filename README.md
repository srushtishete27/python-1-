# Demonstration of Python Data Types

# Integer
num = 10
print("Integer:", num, "| Type:", type(num))

# Float
decimal = 10.5
print("Float:", decimal, "| Type:", type(decimal))

# String
text = "Hello, Python!"
print("String:", text, "| Type:", type(text))

# Boolean
flag = True
print("Boolean:", flag, "| Type:", type(flag))

# List (mutable collection)
my_list = [1, 2, 3, "apple"]
print("List:", my_list, "| Type:", type(my_list))

# Tuple (immutable collection)
my_tuple = (1, 2, 3, "banana")
print("Tuple:", my_tuple, "| Type:", type(my_tuple))

# Dictionary (key-value pairs)
my_dict = {
    "name": "John",
    "age": 20,
    "course": "Python"
}
print("Dictionary:", my_dict, "| Type:", type(my_dict))

# Set (unordered unique elements)
